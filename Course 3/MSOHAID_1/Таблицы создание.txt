-- Таблица "Разработка игрового продукта"
CREATE TABLE Разработка_игрового_продукта (
    Идентификатор_продукта NUMBER PRIMARY KEY,
    Идентификатор_игры NUMBER UNIQUE
);

-- Таблица "Роли"
CREATE TABLE Роли (
    Идентификатор_роли NUMBER PRIMARY KEY,
    Название_роли VARCHAR2(255) UNIQUE
);

-- Таблица "Отделы"
CREATE TABLE Отделы (
    Идентификатор_отдела NUMBER PRIMARY KEY,
    Наименование_отдела VARCHAR2(255) UNIQUE
);

-- Таблица "Вакансии"
CREATE TABLE Вакансии (
    Идентификатор_вакансии NUMBER PRIMARY KEY,
    Должность VARCHAR2(255),
    Отдел VARCHAR2(255) REFERENCES Отделы (Наименование_отдела),
    Описание_вакансии VARCHAR2(1000),
    Заработная_плата FLOAT,
    Статус_вакансии VARCHAR2(255)
);

-- Таблица "Трудовые договоры"
CREATE TABLE Трудовые_договоры (
    Идентификатор_договора NUMBER PRIMARY KEY,
    Дата_начала_действия_договора DATE,
    Дата_окончания_действия_договора DATE,
    Тип_договора VARCHAR2(255)
);

-- Таблица "Сотрудники"
CREATE TABLE Сотрудники (
    Идентификатор_сотрудника NUMBER PRIMARY KEY,
    ФИО_сотрудника VARCHAR2(255) UNIQUE,
    Контракт NUMBER REFERENCES Трудовые_договоры (Идентификатор_договора),
    Роль VARCHAR2(255) REFERENCES Роли (Название_роли),
    Дата_рождения DATE,
    Пол VARCHAR2(255),
    Отдел VARCHAR2(255) REFERENCES Отделы (Наименование_отдела)
);

-- Таблица "Собеседования"
CREATE TABLE Собеседования (
    Идентификатор_собеседования NUMBER PRIMARY KEY,
    Идентификатор_вакансии NUMBER REFERENCES Вакансии (Идентификатор_вакансии),
    Имя_кандидата VARCHAR2(255),
    Дата_собеседования DATE,
    Результат_собеседования VARCHAR2(255),
    Сотрудник_принимавший_собеседование VARCHAR2(255) REFERENCES Сотрудники (ФИО_сотрудника)
);

-- Таблица "Команда разработчиков"
CREATE TABLE Команда_разработчиков (
    Идентификатор_команды NUMBER PRIMARY KEY,
    Название_команды VARCHAR2(255),
    Сотрудники VARCHAR2(255) REFERENCES Сотрудники (ФИО_сотрудника),
    Идентификатор_продукта NUMBER REFERENCES Разработка_игрового_продукта (Идентификатор_продукта)
);

-- Таблица "Банки"
CREATE TABLE Банки (
    Идентификатор_банка NUMBER PRIMARY KEY,
    Наименование_банка VARCHAR2(255) UNIQUE,
    Филиал VARCHAR2(255),
    Местоположение VARCHAR2(255),
    Контактный_телефон VARCHAR2(20)
);

-- Таблица "Счета"
CREATE TABLE Счета (
    Идентификатор_счета NUMBER PRIMARY KEY,
    Наименование_счета VARCHAR2(255) UNIQUE,
    Тип_счета VARCHAR2(255),
    Баланс_счета FLOAT,
    Ответственный_сотрудник VARCHAR2(255) REFERENCES Сотрудники (ФИО_сотрудника),
    Банк VARCHAR2(255) REFERENCES Банки (Наименование_банка)
);

-- Таблица "Транзакции"
CREATE TABLE Транзакции (
    Идентификатор_транзакции NUMBER PRIMARY KEY,
    Дата_транзакции DATE,
    Тип_транзакции VARCHAR2(255),
    Сумма_транзакции FLOAT,
    Описание_транзакции VARCHAR2(1000),
    Банк_одобривший_транзакцию VARCHAR2(255) REFERENCES Банки (Наименование_банка),
    Наименование_счета VARCHAR2(255) REFERENCES Счета (Наименование_счета)
);

-- Таблица "Календарь поставления финансов"
CREATE TABLE Календарь_поставления_финансов (
    Дата_поставки DATE PRIMARY KEY,
    Транзакция NUMBER REFERENCES Транзакции (Идентификатор_транзакции),
    Примечание_к_поставке VARCHAR2(1000),
    Идентификатор_продукта NUMBER REFERENCES Разработка_игрового_продукта (Идентификатор_продукта)
);

-- Таблица "Целевая аудитория"
CREATE TABLE Целевая_аудитория (
    Идентификатор_целевой_аудитории NUMBER PRIMARY KEY,
    Название_целевой_аудитории VARCHAR2(255),
    Описание_целевой_аудитории VARCHAR2(1000)
);

-- Таблица "Каналы продвижения"
CREATE TABLE Каналы_продвижения (
    Идентификатор_канала_продвижения NUMBER PRIMARY KEY,
    Названия_канала_продвижения VARCHAR2(255),
    Описание_канала_продвижения VARCHAR2(1000),
    Бюджет_канала_продвижения DATE REFERENCES Календарь_поставления_финансов (Дата_поставки),
    Сотрудник_занимающийся_продвижением VARCHAR2(255) REFERENCES Сотрудники (ФИО_сотрудника)
);

-- Таблица "Рыночные исследования"
CREATE TABLE Рыночные_исследования (
    Идентификатор_рыночного_исследования NUMBER PRIMARY KEY,
    Тема_рыночного_исследования VARCHAR2(255),
    Анализ_рыночного_исследования VARCHAR2(1000),
    Дата_проведения_рыночного_исследования DATE,
    Сотрудник_проведший_анализ VARCHAR2(255) REFERENCES Сотрудники (ФИО_сотрудника)
);

-- Таблица "Бизнес план продвижения"
CREATE TABLE Бизнес_план_продвижения (
    Идентификатор_задачи NUMBER PRIMARY KEY,
    Название_задачи VARCHAR2(255),
    Описание_задачи VARCHAR2(1000),
    Идентификатор_целевой_аудитории NUMBER REFERENCES Целевая_аудитория (Идентификатор_целевой_аудитории),
    Идентификатор_канала_продвижения NUMBER REFERENCES Каналы_продвижения (Идентификатор_канала_продвижения),
    Анализ_рынка NUMBER REFERENCES Рыночные_исследования (Идентификатор_рыночного_исследования),
    Ответственный_сотрудник VARCHAR2(255),
    Идентификатор_продукта NUMBER REFERENCES Разработка_игрового_продукта (Идентификатор_продукта)
);

-- Таблица "Игроки"
CREATE TABLE Игроки (
    ID NUMBER PRIMARY KEY,
    Имя VARCHAR2(255) UNIQUE,
    Уровень NUMBER,
    Опыт NUMBER
);

-- Таблица "Персонажи"
CREATE TABLE Персонажи (
    ID NUMBER PRIMARY KEY,
    Имя VARCHAR2(255) UNIQUE,
    Владелец_персонажа VARCHAR2(255) REFERENCES Игроки (Имя),
    Описание VARCHAR2(1000)
);

-- Таблица "Инвентарь персонажа"
CREATE TABLE Инвентарь_персонажа (
    ID NUMBER PRIMARY KEY,
    Персонаж VARCHAR2(255) REFERENCES Персонажи (Имя),
    Название VARCHAR2(255),
    Количество NUMBER
);

-- Таблица "Сценарии"
CREATE TABLE Сценарии (
    ID NUMBER PRIMARY KEY,
    Название VARCHAR2(255) UNIQUE,
    Описание VARCHAR2(1000),
    Сценарист VARCHAR2(255) REFERENCES Сотрудники (ФИО_сотрудника),
    Игра NUMBER REFERENCES Разработка_игрового_продукта (ИДЕНТИФИКАТОР_ИГРЫ)
);

-- Таблица "Локации"
CREATE TABLE Локации (
    ID NUMBER PRIMARY KEY,
    Название VARCHAR2(255) UNIQUE,
    Описание VARCHAR2(1000),
    Разработчик VARCHAR2(255) REFERENCES Сотрудники (ФИО_сотрудника)
);

-- Таблица "События"
CREATE TABLE События (
    ID NUMBER PRIMARY KEY,
    Локация VARCHAR2(255) REFERENCES Локации (Название),
    Название VARCHAR2(255),
    Описание VARCHAR2(1000),
    Разработчик VARCHAR2(255)
);

-- Таблица "Задания"
CREATE TABLE Задания (
    ID NUMBER PRIMARY KEY,
    Название VARCHAR2(255) UNIQUE,
    Описание VARCHAR2(1000),
    Награда VARCHAR2(255),
    Сценарий VARCHAR2(255) REFERENCES Сценарии (Название),
    Локация VARCHAR2(255) REFERENCES Локации (Название)
);

-- Таблица "Прогресс игроков в заданиях"
CREATE TABLE Прогресс_игроков_в_заданиях (
    ID NUMBER PRIMARY KEY,
    Игрок VARCHAR2(255) REFERENCES Игроки (Имя),
    Задание VARCHAR2(255) REFERENCES Задания (Название),
    Завершено VARCHAR2(255)
);
